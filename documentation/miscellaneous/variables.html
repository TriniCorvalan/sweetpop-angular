authGuard
Type : CanActivateFn
Default value : (route) => { const auth = inject(AuthService); const router = inject(Router); const session = auth.getSession(); const allowedRoles = route.data['roles'] as UserRole[] | undefined; if (!session) { return router.createUrlTree(['/inicio-sesion']); } if (allowedRoles && !allowedRoles.includes(session.role)) { if (session.role === 'admin') { return router.createUrlTree(['/inventario']); } return router.createUrlTree(['/']); } return true; }
